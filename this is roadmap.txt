Schema::create('users', function (Blueprint $table) {
    $table->id();
    $table->string('name');
    $table->string('email')->unique();
    $table->string('phone')->nullable();
    $table->timestamp('email_verified_at')->nullable();
    $table->string('password');
    $table->string('google_id')->nullable();
    $table->string('avatar')->nullable();
    $table->enum('role', ['super_admin', 'business_owner', 'employee'])->default('business_owner');
    $table->enum('status', ['active', 'inactive', 'suspended'])->default('active');
    $table->rememberToken();
    $table->timestamps();
    $table->softDeletes();
});

// QR Codes Table
Schema::create('qr_codes', function (Blueprint $table) {
    $table->id();
    $table->foreignId('business_id')->constrained()->cascadeOnDelete();
    $table->foreignId('branch_id')->nullable()->constrained()->nullOnDelete();
    $table->foreignId('employee_id')->nullable()->constrained('users')->nullOnDelete();
    $table->string('name'); // e.g., "Table 5", "Room 101", "Reception"

});

// app/Services/QR/QRCodeService.php

namespace App\Services\QR;

use App\Models\QRCode;
use App\Models\Business;
use SimpleSoftwareIO\QrCode\Facades\QrCode;
use Illuminate\Support\Str;
use Illuminate\Support\Facades\Storage;

class QRCodeService
{
    public function generate(Business $business, array $data): QRCode
    {
        $slug = $this->generateUniqueSlug($data['name']);
        $landingUrl = url('/r/' . $slug);
        
        $qrCode = QRCode::create([
            'business_id' => $business->id,
            'branch_id' => $data['branch_id'] ?? null,
            'employee_id' => $data['employee_id'] ?? null,
            'name' => $data['name'],
            'slug' => $slug,
            'type' => $data['type'] ?? 'custom',
            'identifier' => $data['identifier'] ?? null,
            'landing_page_url' => $landingUrl,
            'is_active' => true,
        ]);
        
        // Generate QR Images
        $this->generateQRImages($qrCode, $landingUrl, $data);
        
        return $qrCode->load('business', 'branch', 'employee');
    }
    
    public function generateBulk(Business $business, array $data): array
    {
        $qrCodes = [];
        $prefix = $data['prefix'] ?? 'QR';
        $start = $data['start_number'] ?? 1;
        $end = $data['end_number'] ?? 10;
        
        for ($i = $start; $i <= $end; $i++) {
            $identifier = $prefix . ' ' . $i;
            $qrCodes[] = $this->generate($business, [
                'name' => $identifier,
                'type' => $data['type'] ?? 'custom',
                'identifier' => $identifier,
                'branch_id' => $data['branch_id'] ?? null,
            ]);
        }
        
        return $qrCodes;
    }
    
    public function generateQRImages(QRCode $qrCode, string $url, array $options = [])
    {
        $size = $options['size'] ?? 400;
        $color = $options['color'] ?? '#000000';
        $bgColor = $options['bg_color'] ?? '#ffffff';
        $logoPath = $options['logo'] ?? null;
        
        // PNG Format
        $qrImage = QrCode::format('png')
            ->size($size)
            ->color(0, 0, 0)
            ->backgroundColor(255, 255, 255)
            ->margin(2)
            ->generate($url);
        
        $pngPath = 'qr/' . $qrCode->slug . '.png';
        Storage::put('public/' . $pngPath, $qrImage);
        $qrCode->update(['qr_image_path' => $pngPath]);
        
        // SVG Format
        $qrSvg = QrCode::format('svg')
            ->size($size)
            ->margin(2)
            ->generate($url);
        
        $svgPath = 'qr/' . $qrCode->slug . '.svg';
        Storage::put('public/' . $svgPath, $qrSvg);
        $qrCode->update(['qr_svg_path' => $svgPath]);
        
        // If logo provided, merge logo with QR
        if ($logoPath && Storage::exists('public/' . $logoPath)) {
            $this->mergeLogoWithQR($qrCode, $logoPath);
        }
    }
    
    public function download(QRCode $qrCode, string $format = 'png')
    {
        $path = $format === 'svg' 
            ? $qrCode->qr_svg_path 
            : $qrCode->qr_image_path;
            
        return Storage::disk('public')->download($path, $qrCode->slug . '.' . $format);
    }
    
    public function generatePDF(QRCode $qrCode)
    {
        // Generate printable PDF with QR code
        $pdf = app()->make('dompdf.wrapper');
        $pdf->loadView('qr.print', compact('qrCode'));
        return $pdf->download($qrCode->slug . '.pdf');
    }
    
    private function generateUniqueSlug(string $name): string
    {
        $slug = Str::slug($name) . '-' . Str::random(6);
        while (QRCode::where('slug', $slug)->exists()) {
            $slug = Str::slug($name) . '-' . Str::random(6);
        }
        return $slug;
    }
    
    private function mergeLogoWithQR(QRCode $qrCode, string $logoPath)
    {
        // Use Intervention Image to merge logo
        $qrImage = \Image::make(Storage::path('public/' . $qrCode->qr_image_path));
        $logo = \Image::make(Storage::path('public/' . $logoPath));
        
        $logo->resize(80, 80);
        $qrImage->insert($logo, 'center');
        $qrImage->save(Storage::path('public/' . $qrCode->qr_image_path));
    }
    
    public function recordScan(QRCode $qrCode, array $scanData): QRScan
    {
        $scan = QRScan::create([
            'qr_code_id' => $qrCode->id,
            'business_id' => $qrCode->business_id,
            'session_id' => $scanData['session_id'] ?? session()->getId(),
            'ip_address' => $scanData['ip_address'] ?? request()->ip(),
            'device_type' => $scanData['device_type'] ?? $this->detectDevice(),
            'browser' => $scanData['browser'] ?? $this->detectBrowser(),
            'os' => $scanData['os'] ?? $this->detectOS(),
            'location' => $scanData['location'] ?? null,
            'latitude' => $scanData['latitude'] ?? null,
            'longitude' => $scanData['longitude'] ?? null,
            'scanned_at' => now(),
        ]);
        
        // Increment scan count
        $qrCode->increment('scan_count');
        
        return $scan;
    }
    
    private function detectDevice(): string
    {
        $userAgent = request()->userAgent();
        if (preg_match('/Mobile/i', $userAgent)) return 'mobile';
        if (preg_match('/Tablet/i', $userAgent)) return 'tablet';
        return 'desktop';
    }
    
    private function detectBrowser(): string
    {
        $userAgent = request()->userAgent();
        if (preg_match('/Chrome/i', $userAgent)) return 'Chrome';
        if (preg_match('/Firefox/i', $userAgent)) return 'Firefox';
        if (preg_match('/Safari/i', $userAgent)) return 'Safari';
        return 'Other';
    }
    
    private function detectOS(): string
    {
        $userAgent = request()->userAgent();
        if (preg_match('/Android/i', $userAgent)) return 'Android';
        if (preg_match('/iPhone|iPad/i', $userAgent)) return 'iOS';
        if (preg_match('/Windows/i', $userAgent)) return 'Windows';
        if (preg_match('/Mac/i', $userAgent)) return 'macOS';
        if (preg_match('/Linux/i', $userAgent)) return 'Linux';
        return 'Other';
    }
}

// app/Services/AI/AIService.php

namespace App\Services\AI;

use App\Models\AICredit;
use App\Models\AICreditLog;
use App\Models\Business;
use App\Models\AIReviewTemplate;
use App\Models\Review;
use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Log;

class AIService
{
    private $apiKey;
    private $apiUrl;
    private $model;
    
    public function __construct()
    {
        $this->apiKey = config('services.openai.api_key') ?? config('services.gemini.api_key');
        $this->apiUrl = config('services.openai.api_url', 'https://api.openai.com/v1/chat/completions');
        $this->model = config('services.openai.model', 'gpt-3.5-turbo');
    }
    
    public function generateReviewSuggestions(
        Business $business,
        int $rating,
        string $language = 'en',
        ?string $customPrompt = null
    ): array
    {
        // Check AI Credits
        if (!$this->hasCredits($business)) {
            throw new \Exception('Insufficient AI credits');
        }
        
        // Get template or use default
        $template = $this->getTemplate($business, $rating, $language);
        
        // Build prompt
        $prompt = $customPrompt ?? $this->buildPrompt($business, $template, $rating, $language);
        
        // Call AI API
        $response = $this->callAI($prompt);
        
        // Parse response
        $suggestions = $this->parseSuggestions($response);
        
        // Deduct credits
        $this->deductCredits($business, 1, 'Review generation');
        
        return $suggestions;
    }
    
    public function generateReplyToReview(Review $review): string
    {
        $business = $review->business;
        
        if (!$this->hasCredits($business)) {
            throw new \Exception('Insufficient AI credits');
        }
        
        $prompt = $this->buildReplyPrompt($review);
        $response = $this->callAI($prompt);
        
        $this->deductCredits($business, 1, 'Auto-reply generation');
        
        return $response;
    }
    
    private function buildPrompt(
        Business $business,
        AIReviewTemplate $template,
        int $rating,
        string $language
    ): string
    {
        $languageNames = [
            'en' => 'English',
            'hi' => 'Hindi',
            'gu' => 'Gujarati'
        ];
        
        $languageName = $languageNames[$language] ?? 'English';
        
        return <<<PROMPT
You are helping a customer write a Google review for a business.

Business Details:
- Name: {$business->name}
- Type: {$business->business_type}
- Location: {$business->city}, {$business->state}

Customer's Rating: {$rating} stars

Template: {$template->prompt_template}

Instructions:
1. Generate 3 different review options in {$languageName}
2. Each review should be 20-50 words
3. Reviews should sound natural and genuine
4. Include specific details that would be relevant for this type of business
5. Format your response as a JSON array with 3 objects, each having "text" field

Example format:
[
    {"text": "Review text 1"},
    {"text": "Review text 2"},
    {"text": "Review text 3"}
]

Return ONLY the JSON array, no other text.
PROMPT;

Instructions:

Return ONLY the response text, nothing else.
PROMPT;

    {
        try {
            $response = Http::withHeaders([
                'Authorization' => 'Bearer ' . $this->apiKey,
                'Content-Type' => 'application/json',
            ])->post($this->apiUrl, [
                'model' => $this->model,
                'messages' => [
                    ['role' => 'system', 'content' => 'You are a helpful assistant that generates Google reviews.'],
                    ['role' => 'user', 'content' => $prompt],
                ],
                'temperature' => 0.7,
                'max_tokens' => 500,
            ]);
            
            if ($response->failed()) {
                Log::error('AI API Error', ['response' => $response->body()]);

            Log::error('AI Service Error', ['error' => $e->getMessage()]);

            return $this->getFallbackReviews();
        }

    private function parseSuggestions(string $response): array
    {
        // Try to parse JSON
        $decoded = json_decode($response, true);
        
        if (is_array($decoded)) {
            return array_map(fn($item) => $item['text'] ?? $item, $decoded);
        }
        
        // If not JSON, split by newlines
        $lines = array_filter(explode("\n", $response));
        return array_slice($lines, 0, 3);
    }

namespace App\Services\Payment;

use Razorpay\Api\Api;
use App\Models\Payment;
use App\Models\Subscription;
use App\Models\User;
use App\Models\Plan;
use Exception;

class RazorpayService
{
    private $api;
    
    public function __construct()
    {
        $this->api = new Api(
            config('services.razorpay.key_id'),
            config('services.razorpay.key_secret')
        );
    }
    
    public function createOrder(Plan $plan, User $user, ?string $couponCode = null): array
    {
        $amount = $this->calculateAmount($plan, $couponCode);
        
        $orderData = [
            'receipt' => 'rcpt_' . uniqid(),
            'amount' => $amount * 100, // Convert to paise
            'currency' => 'INR',
            'notes' => [
                'user_id' => $user->id,
                'plan_id' => $plan->id,
                'plan_name' => $plan->name,
            ],
        ];
        
        $order = $this->api->order->create($orderData);
        
        return [
            'order_id' => $order['id'],
            'amount' => $amount,
            'currency' => 'INR',
            'key' => config('services.razorpay.key_id'),
            'name' => config('app.name'),
            'description' => "Subscription: {$plan->name}",
            'prefill' => [
                'name' => $user->name,
                'email' => $user->email,
                'contact' => $user->phone ?? '',
            ],
            'theme' => [
                'color' => '#3399cc',
            ],
        ];
    }
    
    public function createSubscription(Plan $plan, User $user): array
    {
        $subscriptionData = [
            'plan_id' => $plan->razorpay_plan_id,
            'customer_notify' => 1,
            'total_count' => $plan->billing_cycle === 'yearly' ? 1 : 12,
            'notes' => [
                'user_id' => $user->id,
                'plan_id' => $plan->id,
            ],
        ];
        
        $subscription = $this->api->subscription->create($subscriptionData);
        
        return [
            'subscription_id' => $subscription['id'],
            'short_url' => $subscription['short_url'],
        ];
    }
    
    public function verifyPayment(array $data): bool
    {
        try {
            $this->api->utility->verifyPaymentSignature([
                'razorpay_order_id' => $data['razorpay_order_id'],
                'razorpay_payment_id' => $data['razorpay_payment_id'],
                'razorpay_signature' => $data['razorpay_signature'],
            ]);
            
            return true;
        } catch (Exception $e) {
            return false;
        }
    }
    
    public function handleWebhook(array $payload): void
    {
        $event = $payload['event'];
        
        switch ($event) {
            case 'payment.captured':
                $this->handlePaymentCaptured($payload['payload']['payment']['entity']);
                break;
                
            case 'subscription.activated':
                $this->handleSubscriptionActivated($payload['payload']['subscription']['entity']);
                break;
                
            case 'subscription.charged':
                $this->handleSubscriptionCharged($payload['payload']['subscription']['entity']);
                break;
                
            case 'subscription.cancelled':
                $this->handleSubscriptionCancelled($payload['payload']['subscription']['entity']);
                break;
                
            case 'subscription.expired':
                $this->handleSubscriptionExpired($payload['payload']['subscription']['entity']);
                break;
        }
    }
    
    private function handlePaymentCaptured(array $payment)
    {
        Payment::where('razorpay_payment_id', $payment['id'])
            ->update(['status' => 'completed']);
            
        // Activate subscription
        $paymentRecord = Payment::where('razorpay_payment_id', $payment['id'])->first();
        if ($paymentRecord && $paymentRecord->subscription) {
            $paymentRecord->subscription->update([
                'status' => 'active',
                'starts_at' => now(),
            ]);
        }
    }
    
    private function handleSubscriptionActivated(array $subscription)
    {
        Subscription::where('razorpay_subscription_id', $subscription['id'])
            ->update(['status' => 'active']);
    }
    
    private function handleSubscriptionCharged(array $subscription)
    {
        // Extend subscription end date
        $sub = Subscription::where('razorpay_subscription_id', $subscription['id'])->first();
        if ($sub) {
            $sub->update(['ends_at' => now()->addMonth()]);
        }
    }
    
    private function handleSubscriptionCancelled(array $subscription)
    {
        Subscription::where('razorpay_subscription_id', $subscription['id'])
            ->update([
                'status' => 'cancelled',
                'cancelled_at' => now(),
            ]);
    }
    
    private function handleSubscriptionExpired(array $subscription)
    {
        Subscription::where('razorpay_subscription_id', $subscription['id'])
            ->update(['status' => 'expired']);
    }
    
    private function calculateAmount(Plan $plan, ?string $couponCode): float
    {
        $amount = $plan->price;
        
        if ($couponCode) {
            $coupon = \App\Models\Coupon::where('code', $couponCode)
                ->where('is_active', true)
                ->where('valid_from', '<=', now())
                ->where('valid_until', '>=', now())
                ->first();
            
            if ($coupon) {
                if ($coupon->discount_type === 'percentage') {
                    $discount = $amount * ($coupon->discount_value / 100);
                } else {
                    $discount = $coupon->discount_value;
                }
                
                if ($coupon->max_discount) {
                    $discount = min($discount, $coupon->max_discount);
                }
                
                $amount -= $discount;
            }
        }
        
        return max($amount, 0);
    }
}

    public function sendReviewRequest(

        string $customerPhone,
        string $customerName = null,
        string $qrSlug = null
    ): WhatsAppMessage
    {
        $reviewLink = $qrSlug ? url('/r/' . $qrSlug) : $business->google_review_link;
        
        $message = $this->buildReviewRequestMessage($business, $customerName, $reviewLink);
        
        $messageRecord = WhatsAppMessage::create([
            'business_id' => $business->id,
            'customer_phone' => $this->formatPhone($customerPhone),
            'customer_name' => $customerName,
            'message_type' => 'review_request',
            'message_content' => $message,
            'status' => 'pending',
        ]);
        
        try {
            $response = $this->sendWhatsAppMessage($customerPhone, $message);
            
            $messageRecord->update([
                'status' => 'sent',
                'whatsapp_message_id' => $response['messages'][0]['id'] ?? null,
                'response' => $response,
                'sent_at' => now(),
            ]);
        } catch (\Exception $e) {
            Log::error('WhatsApp Send Error', ['error' => $e->getMessage()]);
            $messageRecord->update(['status' => 'failed']);
        }
        
        return $messageRecord;
    }
    
    public function sendBulkReviewRequests(
        Business $business,
        array $customers,
        string $messageType = 'review_request'
    ): array
    {
        $results = [];
        
        foreach ($customers as $customer) {
            $results[] = $this->sendReviewRequest(
                $business,
                $customer['phone'],
                $customer['name'] ?? null,
                $customer['qr_slug'] ?? null
            );
            
            // Add delay to avoid rate limiting
            usleep(200000); // 200ms delay
        }
        
        return $results;
    }
    
    public function sendTemplateMessage(
        string $phone,
        string $templateName,
        array $parameters = []
    ): array
    {
        $components = [];
        
        if (!empty($parameters)) {
            $components[] = [
                'type' => 'body',
                'parameters' => array_map(fn($p) => ['type' => 'text', 'text' => $p], $parameters),
            ];
        }
        
        $payload = [
            'messaging_product' => 'whatsapp',
            'to' => $this->formatPhone($phone),
            'type' => 'template',
            'template' => [
                'name' => $templateName,
                'language' => ['code' => 'en'],
                'components' => $components,
            ],
        ];
        
        return $this->apiCall($payload);
    }
    
    private function buildReviewRequestMessage(
        Business $business,
        ?string $customerName,
        string $reviewLink
    ): string
    {
        $greeting = $customerName ? "Hi {$customerName}" : "Hello";
        
        return <<<MESSAGE
{$greeting}! 👋

Thank you for visiting {$business->name}!

We'd love to hear about your experience. Your feedback helps us serve you better.

⭐ Click the link below to leave a quick review:
{$reviewLink}

It takes just 30 seconds! 🙏

Thank you for your support!
- Team {$business->name}
MESSAGE;
    }
    
    private function sendWhatsAppMessage(string $phone, string $message): array
    {
        $payload = [
            'messaging_product' => 'whatsapp',
            'to' => $this->formatPhone($phone),
            'type' => 'text',
            'text' => ['body' => $message],
        ];
        
        return $this->apiCall($payload);
    }
    
    private function apiCall(array $payload): array
    {
        $response = Http::withHeaders([
            'Authorization' => 'Bearer ' . $this->accessToken,
            'Content-Type' => 'application/json',
        ])->post($this->apiUrl . '/' . $this->phoneNumberId . '/messages', $payload);
        
        if ($response->failed()) {
            throw new \Exception('WhatsApp API error: ' . $response->body());
        }
        
        return $response->json();
    }
    
    private function formatPhone(string $phone): string
    {
        // Remove all non-digit characters
        $phone = preg_replace('/\D/', '', $phone);
        
        // Add India country code if not present
        if (strlen($phone) === 10) {
            $phone = '91' . $phone;
        }
        
        return $phone;
    }
    
    public function handleWebhook(array $payload): void
    {
        if (isset($payload['entry'][0]['changes'][0]['value']['statuses'])) {
            foreach ($payload['entry'][0]['changes'][0]['value']['statuses'] as $status) {
                $this->updateMessageStatus(
                    $status['id'],
                    $status['status']
                );
            }
        }
    }
    
    private function updateMessageStatus(string $messageId, string $status)
    {
        WhatsAppMessage::where('whatsapp_message_id', $messageId)
            ->update(['status' => $status]);
    }
}

class AnalyticsService
{
    public function getDashboardStats(Business $business): array
    {
        $totalScans = QRScan::where('business_id', $business->id)->count();
        $totalReviews = Review::where('business_id', $business->id)->count();
        $avgRating = Review::where('business_id', $business->id)->avg('rating') ?? 0;
        $conversionRate = $totalScans > 0 ? ($totalReviews / $totalScans) * 100 : 0;
        
        // Monthly growth
        $lastMonthScans = QRScan::where('business_id', $business->id)
            ->whereBetween('scanned_at', [
                now()->subMonth()->startOfMonth(),
                now()->subMonth()->endOfMonth()
            ])->count();
            
        $thisMonthScans = QRScan::where('business_id', $business->id)
            ->whereBetween('scanned_at', [
                now()->startOfMonth(),
                now()->endOfMonth()
            ])->count();
            
        $monthlyGrowth = $lastMonthScans > 0 
            ? (($thisMonthScans - $lastMonthScans) / $lastMonthScans) * 100 
            : 0;
        
        return [
            'total_scans' => $totalScans,
            'total_reviews' => $totalReviews,
            'avg_rating' => round($avgRating, 1),
            'conversion_rate' => round($conversionRate, 2),
            'monthly_growth' => round($monthlyGrowth, 2),
            'this_month_scans' => $thisMonthScans,
            'this_month_reviews' => Review::where('business_id', $business->id)
                ->whereBetween('created_at', [now()->startOfMonth(), now()->endOfMonth()])
                ->count(),
        ];
    }
    
    public function getScanAnalytics(Business $business, string $period = 'daily'): array
    {
        $query = QRScan::where('business_id', $business->id);
        
        $groupBy = match($period) {
            'hourly' => DB::raw('DATE_FORMAT(scanned_at, "%Y-%m-%d %H:00")'),
            'daily' => DB::raw('DATE(scanned_at)'),
            'weekly' => DB::raw('YEARWEEK(scanned_at)'),
            'monthly' => DB::raw('DATE_FORMAT(scanned_at, "%Y-%m")'),
            default => DB::raw('DATE(scanned_at)'),
        };
        
        $data = $query->select([
                DB::raw($groupBy . ' as period'),
                DB::raw('COUNT(*) as scans'),
                DB::raw('SUM(CASE WHEN converted_to_review = 1 THEN 1 ELSE 0 END) as reviews'),
            ])
            ->groupBy('period')
            ->orderBy('period')
            ->get();
        
        return [
            'labels' => $data->pluck('period')->toArray(),
            'scans' => $data->pluck('scans')->toArray(),
            'reviews' => $data->pluck('reviews')->toArray(),
        ];
    }
    
    public function getDeviceAnalytics(Business $business): array
    {
        return QRScan::where('business_id', $business->id)
            ->select('device_type', DB::raw('COUNT(*) as count'))
            ->groupBy('device_type')
            ->get()
            ->toArray();
    }
    
    public function getBranchWiseAnalytics(Business $business): array
    {
        if (!$business->branches()->exists()) {
            return [];
        }
        
        $branches = $business->branches()->withCount([
            'qrCodes as total_scans' => function ($q) {
                $q->join('qr_scans', 'qr_codes.id', '=', 'qr_scans.qr_code_id')
                  ->select(DB::raw('COUNT(*)'));
            },
        ])->with(['reviews' => function ($q) {
            $q->select('branch_id', 
                DB::raw('COUNT(*) as count'),
                DB::raw('AVG(rating) as avg_rating')
            )->groupBy('branch_id');
        }])->get();
        
        return $branches->map(function ($branch) {
            return [
                'branch_id' => $branch->id,
                'branch_name' => $branch->name,
                'total_scans' => $branch->total_scans,
                'total_reviews' => $branch->reviews->sum('count'),
                'avg_rating' => $branch->reviews->avg('avg_rating') ?? 0,
            ];
        })->toArray();
    }
    
    public function getEmployeeWiseAnalytics(Business $business): array
    {
        return $business->employees()->with(['qrCodes' => function ($q) {
            $q->withCount('scans', 'reviews');
        }])->get()->map(function ($employee) {
            $totalScans = $employee->qrCodes->sum('scans_count');
            $totalReviews = $employee->qrCodes->sum('reviews_count');
            
            return [
                'employee_id' => $employee->id,
                'employee_name' => $employee->name,
                'total_scans' => $totalScans,
                'total_reviews' => $totalReviews,
                'conversion_rate' => $totalScans > 0 ? round(($totalReviews / $totalScans) * 100, 2) : 0,
                'avg_rating' => Review::where('employee_id', $employee->id)->avg('rating') ?? 0,
            ];
        })->toArray();
    }
    
    public function getRatingDistribution(Business $business): array
    {
        return Review::where('business_id', $business->id)
            ->select('rating', DB::raw('COUNT(*) as count'))
            ->groupBy('rating')
            ->orderBy('rating', 'desc')
            ->get()
            ->toArray();
    }
    
    public function getConversionFunnel(Business $business): array
    {
        $totalScans = QRScan::where('business_id', $business->id)->count();
        $ratingSelected = QRScan::where('business_id', $business->id)
            ->where('converted_to_review', true)->count();
        $reviewSubmitted = Review::where('business_id', $business->id)
            ->where('status', 'published')->count();
        
        return [
            [
                'step' => 'QR Scanned',
                'count' => $totalScans,
                'percentage' => 100,
            ],
            [
                'step' => 'Rating Selected',
                'count' => $ratingSelected,
                'percentage' => $totalScans > 0 ? round(($ratingSelected / $totalScans) * 100, 2) : 0,
            ],
            [
                'step' => 'Review Published',
                'count' => $reviewSubmitted,
                'percentage' => $totalScans > 0 ? round(($reviewSubmitted / $totalScans) * 100, 2) : 0,
            ],
        ];
    }
}

    public function reviewPage(string $slug)
    {
        $qrCode = QRCode::where('slug', $slug)
            ->where('is_active', true)
            ->with('business')
            ->firstOrFail();
        
        // Record scan
        $scanService = app(QRCodeService::class);
        $scan = $scanService->recordScan($qrCode, [
            'ip_address' => request()->ip(),
        ]);
        
        session(['current_scan_id' => $scan->id]);
        
        return view('landing.review-page', [
            'qrCode' => $qrCode,
            'business' => $qrCode->business,
            'scan' => $scan,
        ]);
    }
    
    public function generateReview(Request $request, string $slug)
    {
        $request->validate([
            'rating' => 'required|integer|min:1|max:5',
            'language' => 'nullable|in:en,hi,gu',
        ]);
        
        $qrCode = QRCode::where('slug', $slug)->firstOrFail();
        $aiService = app(AIService::class);
        
        try {
            $suggestions = $aiService->generateReviewSuggestions(
                $qrCode->business,
                $request->rating,
                $request->language ?? 'en'
            );
            
            // Mark scan as converted
            if ($scanId = session('current_scan_id')) {
                QRScan::where('id', $scanId)->update(['converted_to_review' => true]);
                $qrCode->increment('review_count');
            }
            
            return response()->json([
                'success' => true,
                'suggestions' => $suggestions,
            ]);
            
        } catch (\Exception $e) {
            return response()->json([
                'success' => false,
                'message' => $e->getMessage(),
            ], 400);
        }
    }
    
    public function submitReview(Request $request, string $slug)
    {
        $request->validate([
            'rating' => 'required|integer|min:1|max:5',
            'review_text' => 'required|string|max:1000',
            'customer_name' => 'nullable|string|max:100',
        ]);
        
        $qrCode = QRCode::where('slug', $slug)->firstOrFail();
        
        // Save review
        $review = $qrCode->reviews()->create([
            'business_id' => $qrCode->business_id,
            'branch_id' => $qrCode->branch_id,
            'scan_id' => session('current_scan_id'),
            'rating' => $request->rating,
            'review_text' => $request->review_text,
            'customer_name' => $request->customer_name,
            'source' => 'qr_scan',
            'status' => 'pending',
        ]);
        
        // Get Google review link
        $googleLink = $qrCode->business->google_review_link;
        
        return response()->json([
            'success' => true,
            'review_id' => $review->id,
            'google_review_link' => $googleLink,
        ]);
    }
    
    public function markReviewPublished(Request $request, string $slug)
    {
        $request->validate([
            'review_id' => 'required|exists:reviews,id',
        ]);
        
        $review = \App\Models\Review::find($request->review_id);
        $review->update([
            'status' => 'published',
        ]);
        
        // Track customer for loyalty
        $this->trackCustomer($review);
        
        return response()->json(['success' => true]);
    }
    
    private function trackCustomer($review)
    {
        // Check if customer exists
        $customer = \App\Models\Customer::where('business_id', $review->business_id)
            ->where('phone', $review->customer_phone)
            ->first();
            
        if ($customer) {
            $customer->increment('visit_count');
            $customer->increment('review_count');
            $customer->update([
                'last_visit' => now(),
                'is_returning' => true,
            ]);
        } else {
            \App\Models\Customer::create([
                'business_id' => $review->business_id,
                'phone' => $review->customer_phone,
                'email' => $review->customer_email,
                'name' => $review->customer_name,
                'visit_count' => 1,
                'review_count' => 1,
                'first_visit' => now(),
                'last_visit' => now(),
            ]);
        }
    }

namespace App\Http\Middleware;

use Closure;
use App\Models\Subscription;
use Illuminate\Support\Facades\Auth;

class CheckSubscription
{
    public function handle($request, Closure $next, $feature = null)
    {
        $user = Auth::user();
        $business = $user->business;
        
        if (!$business) {
            return redirect()->route('onboarding');
        }
        
        $subscription = $user->activeSubscription;
        
        if (!$subscription) {
            return redirect()->route('plans.index')->with('error', 'Please subscribe to a plan.');
        }
        
        // Check specific feature access
        if ($feature) {
            $limits = $subscription->limits ?? [];
            
            switch ($feature) {
                case 'whatsapp':
                    if (!($limits['whatsapp'] ?? false)) {
                        return redirect()->route('plans.index')->with('error', 'Upgrade to Premium for WhatsApp integration.');
                    }
                    break;
                    
                case 'sms':
                    if (!($limits['sms'] ?? false)) {
                        return redirect()->route('plans.index')->with('error', 'Upgrade your plan for SMS campaigns.');
                    }
                    break;
                    
                case 'nfc':
                    if (!($limits['nfc'] ?? false)) {
                        return redirect()->route('plans.index')->with('error', 'Upgrade to Premium for NFC cards.');
                    }
                    break;
                    
                case 'multi_branch':
                    if (!($limits['branches'] > 1 ?? false)) {
                        return redirect()->route('plans.index')->with('error', 'Upgrade to Enterprise for multiple branches.');
                    }
                    break;
            }
        }
        
        return $next($request);
    }
}

namespace App\Http\Middleware;

use Closure;
use App\Models\QRCode;

class CheckQRLimit
{
    public function handle($request, Closure $next)
    {
        $user = Auth::user();
        $subscription = $user->activeSubscription;
        $limits = $subscription->limits ?? [];
        $qrLimit = $limits['qr_codes'] ?? 1;
        
        // -1 means unlimited
        if ($qrLimit !== -1) {
            $currentCount = QRCode::where('business_id', $user->business->id)->count();
            
            if ($currentCount >= $qrLimit) {
                return redirect()->route('qr-codes.index')
                    ->with('error', "You've reached your QR code limit ({$qrLimit}). Upgrade your plan for more.");
            }
        }
        
        return $next($request);
    }
}

// app/Console/Kernel.php

protected function schedule(Schedule $schedule)
{
    // Reset monthly review limits
    $schedule->command('subscription:reset-monthly-limits')
        ->monthlyOn(1, '00:00');
    
    // Reset AI credits monthly
    $schedule->command('ai:reset-credits')
        ->monthlyOn(1, '00:00');
    
    // Send review reminders
    $schedule->command('reminders:send-pending')
        ->everyFifteenMinutes();
    
    // Check expired subscriptions
    $schedule->command('subscription:check-expired')
        ->dailyAt('00:00');
    
    // Update employee performance
    $schedule->command('analytics:update-employee-performance')
        ->dailyAt('01:00');
    
    // Clean up old scan data (older than 1 year for free users)
    $schedule->command('analytics:cleanup-old-data')
        ->monthlyOn(15, '02:00');
    
    // Sync Google reviews (if implemented)
    $schedule->command('google:sync-reviews')
        ->hourly();
}
